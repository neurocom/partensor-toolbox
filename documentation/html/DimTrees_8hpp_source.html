541 template<typename Array>
542 void Create( std::array<int,RootExprNode::LabelSetSize> &aLabelSet,
543 Array const &aTnsDims,
544 int const R,
545 Tensor<static_cast<int>(TnsSize)> const &aTnsX )
546 {
547 constexpr bool IsLeaf = RootExprNode::IsLeaf;
548 constexpr bool IsRoot = RootExprNode::IsRoot;
549
550 static_assert(! (IsLeaf && IsRoot), "Tree expression with 1 dimension!" );
551
552 constexpr std::size_t vHalf = RootExprNode::DIM_HALF_SIZE;
553 constexpr std::size_t vLabelSetSize = RootExprNode::LabelSetSize;
554
555 root.mUpdated = true;
556 root.mTnsX = aTnsX;
557 std::copy(aTnsDims.begin(), aTnsDims.begin()+TnsSize, root.mTnsDims.begin());
558 root.mLabelSet = aLabelSet;
559 root.mGramian.setConstant(1);
560
561 // Create left child
562 std::copy(aLabelSet.begin(), aLabelSet.begin()+vHalf, root.left.mLabelSet.begin());
563 std::copy(aLabelSet.end()-(TnsSize-vHalf), aLabelSet.end(), root.left.mDeltaSet.begin());
564 std::copy(aTnsDims.begin(), aTnsDims.begin()+vHalf, root.left.mTnsDims.begin()+1);
565 root.left.mTnsDims.front() = R;
566 RandomTensorGen(root.left.mTnsDims, root.left.mTnsX);
567
568 // Create right child
569 std::copy(aLabelSet.end()-(TnsSize-vHalf), aLabelSet.end(), root.right.mLabelSet.begin());
570 std::copy(aLabelSet.begin(), aLabelSet.begin()+vHalf, root.right.mDeltaSet.begin());
571 std::copy(aTnsDims.begin()+vHalf, aTnsDims.begin()+vLabelSetSize, root.right.mTnsDims.begin()+1);
572 root.right.mTnsDims.front() = R;
573 RandomTensorGen(root.right.mTnsDims, root.right.mTnsX);
574
575 // Set key in root node
576 root.mKey = root.left.mLabelSet[vHalf-1];
577
578 Create(root.left.mTnsDims, R, root.left);
579 Create(root.right.mTnsDims, R, root.right);
580 }
