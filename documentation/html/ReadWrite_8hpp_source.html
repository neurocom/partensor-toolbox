678 template<std::size_t TnsSize>
679 void offset_calculation( std::array<int, TnsSize> const &dimensions,
680 std::array<int, TnsSize> const &skip,
681 std::array<int, TnsSize> &offset )
682 {
683 for (int i=TnsSize-1; i>1; i--)
684 {
685 // How many cubes to skip
686 offset[i] = std::accumulate(dimensions.begin(), dimensions.end()-(TnsSize-i), 1, std::multiplies<int>()) * skip[i];
687 }
688 offset[1] = dimensions[0] * skip[1]; // how many matrices to skip
689 offset[0] = skip[0]; // how many rows to skip
690 }

713 template<std::size_t idx, std::size_t TnsSize>
714 void readTensor_util( std::array<int, TnsSize> const &tnsDims,
715 std::array<int, TnsSize> const &local_tnsDims,
716 std::array<int, TnsSize> const &offset,
717 std::ifstream &is,
718 Tensor<static_cast<int>(TnsSize)> &dat )
719 {
720 using DT = typename TensorTraits<Tensor<static_cast<int>(TnsSize)>>::DataType;
721 if constexpr (idx>0)
722 {
723 // e.g. I*J*K in 4D tensor
724 int dim_product = std::accumulate(tnsDims.begin(), tnsDims.begin()+idx, 1, std::multiplies<int>());
725 // ignore
726 is.ignore(offset[idx]*sizeof(DT));
727
728 for (int i=0; i<local_tnsDims[idx]; i++)
729 {
730 readTensor_util<idx-1, TnsSize>(tnsDims, local_tnsDims, offset, is, dat);
731 }
732 is.ignore( dim_product*(tnsDims[idx] - local_tnsDims[idx] - offset[idx])*sizeof(DT) );
733 }
734 else
735 {
736 Tensor<1> tns_col(local_tnsDims[0]);
737 // ignore offset[0] rows
738 is.ignore(offset[0]*sizeof(DT));
739 is.read(reinterpret_cast<char*>(tns_col.data()), local_tnsDims[0]*sizeof(DT));
740 // use tensor indexing for reading from file
741 for (int i=0; i<local_tnsDims[0]; i++)
742 {
743 static int idx_read = 0;
744 dat(idx_read) = tns_col(i);
745
746 idx_read++;
747 }
748 // ignore remaining column elements
749 is.ignore( (tnsDims[0] - local_tnsDims[0] - offset[0])*sizeof(DT) );
750 }
751 }
