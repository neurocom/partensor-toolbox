301 /*
302 * @brief Combine the values stored by each process into a single
303 * value at the root.
304 *
305 * @c reduce is a collective algorithm that combines the values
306 * stored by each process into a single value at the @c root. The
307 * values can be combined arbitrarily, specified via a function
308 * object. The type @c T of the values may be any type that is
309 * serializable or has an associated MPI data type. One can think of
310 * this operation as a @c gather to the @p root, followed by an @c
311 * std::accumulate() over the gathered values and using the operation
312 * @c op.
313 *
314 * When the type @c T has an associated MPI data type, this routine
315 * invokes @c MPI_Reduce to perform the reduction. If possible,
316 * built-in MPI operations will be used; otherwise, @c reduce() will
317 * create a custom MPI_Op for the call to MPI_Reduce.
318 *
319 * @param comm [in] The communicator over which the reduction will
320 * occur.
321 *
322 * @param in_values [in] The local value to be combined with the local
323 * values of every other process. For reducing arrays, @c in_values
324 * contains a pointer to the local values. In this case, @c n is
325 * the number of values that will be reduced. Reduction occurs
326 * independently for each of the @p n values referenced by @p
327 * in_values, e.g., calling reduce on an array of @p n values is
328 * like calling @c reduce @p n separate times, one for each
329 * location in @p in_values and @p out_values.
330 *
331 * @param out_values [in,out] Will receive the result of the reduction
332 * operation, but only for the @p root process. Non-root processes
333 * may omit if parameter; if they choose to supply the parameter,
334 * it will be unchanged. For reducing arrays, @c out_values
335 * contains a pointer to the storage for the output values.
336 *
337 * @param op [in] The binary operation that combines two values of type
338 * @c T into a third value of type @c T. For types @c T that has
339 * associated MPI data types, @c op will either be translated into
340 * an @c MPI_Op (via @c MPI_Op_create) or, if possible, mapped
341 * directly to a built-in MPI operation. See @c is_mpi_op in the @c
342 * operations.hpp header for more details on this mapping. For any
343 * non-built-in operation, commutativity will be determined by the
344 * @c is_commmutative trait (also in @c operations.hpp): users are
345 * encouraged to mark commutative operations as such, because it
346 * gives the implementation additional latitude to optimize the
347 * reduction operation.
348 *
349 * @param root [in] The process ID number that will receive the final,
350 * combined value. This value must be the same on all processes.
351 */
352 template<typename T, typename Op>
353 void reduce(const cartesian_communicator &comm,
354 const T* in_values,
355 int n,
356 T* out_values,
357 Op op,
358 int root )
359 {
360 boost::mpi::reduce(static_cast<const Boost_CartCommunicator &>(comm), in_values, n, out_values, op, root );
361 }

393 /*
394 * @brief Gather the values stored at every process into a vector of
395 * values from each process.
396 *
397 * @c all_gatherv is a collective algorithm that collects the values
398 * stored at each process into a vector of values at each
399 * process. This vector is indexed by the process number that the
400 * value came from. The type @c T of the values may be any type that
401 * is serializable or has an associated MPI data type.
402 *
403 * @param comm The communicator over which the gather will occur.
404 *
405 * @param in_values The array of values to be transmitted by each process.
406 *
407 * @param in_size For each process this specifies the size of @p in_values.
408 *
409 * @param out_values A pointer to storage that will be populated with
410 * the values from each process.
411 *
412 * @param sizes A vector containing the number of elements each
413 * process will send.
414 *
415 * @param displs A vector such that the i-th entry specifies the
416 * displacement (relative to @p out_values) from which to take the ingoing
417 * data at the @p root process. Overloaded versions for which @p displs is
418 * omitted assume that the data is to be placed contiguously at each process.
419 *
420 */
421 template<typename T>

445 template<std::size_t _TnsSize>
446 void create_layer_grid( cartesian_communicator &grid,
447 std::vector<cartesian_communicator> &layer_comm,
448 std::array<int, _TnsSize> &layer_rank )
449 {
450 std::vector<int> layer_dims(_TnsSize-1);
451 std::array<int, _TnsSize> free_coords;
452
453 for (std::size_t i = 0; i < _TnsSize; ++i)
454 {
455 std::fill(free_coords.begin(), free_coords.end(), 1);
456 free_coords[i] = 0;
457 int pos = 0;
458 for (std::size_t j = 0; j < _TnsSize; ++j) {
459 if (free_coords[j]) {
460 layer_dims[pos++] = free_coords[j] * j;
461 }
462 }
463 // create the sub communicator in the cartesian communicator for layers
464 layer_comm.push_back(cartesian_communicator(grid, layer_dims));
465 // ID for each processor in layers sub communicator
466 layer_rank[i] = layer_comm[i].rank();
467 }
468 }
