142  /*
143  * Compute the cost function value at the end of each outer iteration
144  * based on the last factor.
145  *
146  * @param mv [in] Struct where ALS variables are stored.
147  * @param st [in,out] Struct where the returned values of @c CpdDimTree are stored.
148  * In this case the cost function value is updated.
149  */
150  void cost_function ( Member_Variables const &mv,
151  Status &st)
152  {
153  st.f_value = sqrt( st.frob_tns -2 * ((mv.mttkrp * mv.currentFactor.transpose()).trace()) + (mv.cwise_factor_product * mv.last_gramian).trace() );
154  }

156  /*
157  * Compute the cost function value at the end of each outer iteration
158  * based on the last accelerated factor.
159  *
160  * @param mv [in] Struct where ALS variables are stored.
161  * @param st [in] Struct where the returned values of @c CpdDimTree
162  * are stored.
163  * @param factors [in] Accelerated factors.
164  * @param factors_T_factors [in] Gramian matrices of factors.
165  *
166  * @returns The cost function calculated with the accelerated factors.
167  */
168  double accel_cost_function(Member_Variables const &mv,
169  Status const &st,
170  MatrixArray const &factors,
171  MatrixArray const &factors_T_factors)
172  {
173  return sqrt (st.frob_tns + (PartialCwiseProd(factors_T_factors, lastFactor) * factors_T_factors[lastFactor]).trace()
174  - 2 * ((PartialKhatriRao(factors, lastFactor).transpose() * mv.tnsX_mat_lastFactor_T) * factors[lastFactor]).trace() );
175  }

177  /*
178  * Based on each factor's constraint, a different
179  * update function is used at every outer iteration.
180  *
181  * Computes also factor^T * factor at the end.
182  *
183  * @tparam Dimensions Array type containing the Tensor dimensions.
184  *
185  * @param idx [in] Factor to be updated.
186  * @param R [in] The rank of decomposition.
187  * @param tnsDims [in] Tensor Dimensions. Each index contains the corresponding factor's rows length.
188  * @param st [in] Struct where the returned values of @c CpdDimTree are stored.
189  * @param mv [in,out] Struct where ALS variables are stored.
190  * Updates the current factor (@c Matrix type) and then updates
191  * the same factor of @c FactorDimTree type.
192  */
193  template<typename Dimensions>
194  void update_factor(int const idx,
195  std::size_t const R,
196  Dimensions const &tnsDims,
197  Status const &st,
198  Member_Variables &mv )
199  {
200  switch ( st.options.constraints[idx] )
201  {
202  case Constraint::unconstrained:
203  {
204  mv.currentFactor = mv.mttkrp * mv.cwise_factor_product.inverse();
205  break;
206  }
207  case Constraint::nonnegativity:
208  {
209  mv.temp_matrix = mv.currentFactor;
210  NesterovMNLS(mv.cwise_factor_product, mv.mttkrp, st.options.nesterov_delta_1,
211  st.options.nesterov_delta_2, mv.currentFactor);
212  if(mv.currentFactor.cwiseAbs().colwise().sum().minCoeff() == 0)
213  mv.currentFactor = 0.9 * mv.currentFactor + 0.1 * mv.temp_matrix;
214  break;
215  }
216  case Constraint::orthogonality:
217  {
218  mv.temp_matrix = mv.mttkrp.transpose() * mv.mttkrp;
219  Eigen::SelfAdjointEigenSolver<Matrix> eigensolver(mv.temp_matrix);
220  mv.temp_matrix.noalias() = (eigensolver.eigenvectors())
221  * (eigensolver.eigenvalues().cwiseInverse().cwiseSqrt().asDiagonal())
222  * (eigensolver.eigenvectors().transpose());
223  mv.currentFactor.noalias() = mv.mttkrp * mv.temp_matrix;
224  break;
225  }
226  case Constraint::sparsity:
227  break;
228  default: // in case of Constraint::constant
229  break;
230  }
231  mv.it_factor->factor = matrixToTensor(mv.currentFactor, tnsDims[idx], static_cast<int>(R)); // Map factor from Eigen Matrix to Eigen Tensor
232  mv.it_factor->gramian = (mv.it_factor->factor).contract(mv.it_factor->factor, mv.product_dims); // Compute Covariance Tensor
233  }

235  /*
236  * @brief Line Search Acceleration
237  *
238  * Performs an acceleration step in the updated factors, and keeps the accelerated factors when
239  * the step succeeds. Otherwise, the acceleration step is ignored.
240  * Line Search Acceleration reduces the number outer iterations in the ALS algorithm.
241  *
242  * @note This implementation ONLY, if factors are of @c Matrix type.
243  *
244  * @tparam Dimensions Array type containing the Tensor dimensions.
245  *
246  * @param mv [in,out] Struct where ALS variables are stored.
247  * In case the acceration is successful factor and
248  * factor^T * factor of @c FactorDimTree type are updated.
249  * @param st [in,out] Struct where the returned values of @c CpdDimTree are stored.
250  * If the acceleration succeeds updates the cost function value.
251  *
252  */
253  template<typename Dimensions>
254  void line_search_accel(Dimensions const &tnsDims,
255  std::size_t const R,
256  Member_Variables &mv,
257  Status &st)
258  {
259  double f_accel = 0.0; // Objective Value after the acceleration step
260  double const accel_step = pow(st.ao_iter+1,(1.0/(st.options.accel_coeff)));
261 
262  Matrix factor;
263  Matrix old_factor;
264  MatrixArray accel_factors;
265  MatrixArray accel_gramians;
266 
267  for(std::size_t i=0; i<TnsSize; ++i)
268  {
269  factor = tensorToMatrix(mv.factors[i].factor, tnsDims[i],static_cast<int>(R));
270  old_factor = tensorToMatrix(mv.old_factors[i].factor, tnsDims[i],static_cast<int>(R));
271  accel_factors[i] = old_factor + accel_step * (factor - old_factor);
272  accel_gramians[i] = accel_factors[i].transpose() * accel_factors[i];
273 
274  mv.it_factor++;
275  mv.it_old_factor++;
276  }
277 
278  f_accel = accel_cost_function(mv, st, accel_factors, accel_gramians);
279  if (st.f_value > f_accel)
280  {
281  for(std::size_t i=0; i<TnsSize; ++i)
282  {
283  mv.factors[i].factor = matrixToTensor(accel_factors[i], tnsDims[i], static_cast<int>(R));
284  mv.factors[i].gramian = matrixToTensor(accel_gramians[i], static_cast<int>(R), static_cast<int>(R));
285  }
286  st.f_value = f_accel;
287  Partensor()->Logger()->info("Acceleration Step SUCCEEDED! at iter: {}", st.ao_iter);
288  }
289  else
290  st.options.accel_fail++;
291 
292  if (st.options.accel_fail==5)
293  {
294  st.options.accel_fail=0;
295  st.options.accel_coeff++;
296  }
297  }

299  /*
300  * Sequential implementation of Alternating Least Squares (ALS) method
301  * with Dimension Trees.
302  *
303  * @tparam Dimensions Array type containing the Tensor dimensions.
304  *
305  * @param tnsDims [in] Tensor Dimensions. Each index contains the corresponding factor's rows length.
306  * @param R [in] The rank of decomposition.
307  * @param mv [in] Struct where ALS variables are stored and being updated
308  * until a termination condition is true.
309  * @param st [in,out] Struct where the returned values of @c CpdDimTree are stored.
310  */
311  template<typename Dimensions>
312  void als(Dimensions const &tnsDims,
313  std::size_t const R,
314  Member_Variables &mv,
315  Status &status)
316  {
317  if (status.options.acceleration)
318  {
319  mv.tnsX_mat_lastFactor_T = (Matricization(mv.tnsX, lastFactor)).transpose();
320  }
321 
322  if(status.options.normalization)
323  {
324  choose_normilization_factor(status, mv.all_orthogonal, mv.weight_factor);
325  }
326 
327  mv.tnsX_approx.resize(tnsDims);
328  CpdGen(mv.factors, R, mv.tnsX_approx);
329 
330  status.frob_tns = square_norm(mv.tnsX);
331  status.f_value = norm(mv.tnsX - mv.tnsX_approx); // Error_tnsX
332  status.rel_costFunction = status.f_value/sqrt(status.frob_tns);
333  // increments from 1 to labelSet.size()
334  std::iota(mv.labelSet.begin(), mv.labelSet.end(), 1);
335 
336  ExprTree<TnsSize> tree;
337  tree.Create(mv.labelSet, tnsDims, R, mv.tnsX);
338 
339  mv.it_factor = mv.factors.begin();
340  for(std::size_t k = 0; k<TnsSize; k++)
341  {
342  mv.it_factor->leaf = static_cast<TnsNode<1>*>(search_leaf(k+1, tree));
343  mv.it_factor++;
344  }
345 
346  // ---- Loop until ALS converges ----
347  while(1)
348  {
349  status.ao_iter++;
350  mv.it_factor = mv.factors.begin();
351  Partensor()->Logger()->info("iter: {} -- fvalue: {} -- relative_costFunction: {}", status.ao_iter,
352  status.f_value, status.rel_costFunction);
353 
354  for(std::size_t i=0; i<TnsSize; i++)
355  {
356  mv.it_factor->leaf->UpdateTree(TnsSize, i, mv.it_factor);
357 
358  // Maps from Eigen Tensor to Eigen Matrix
359  mv.temp_matrix = tensorToMatrix(*reinterpret_cast<TensorMatrixType *>(mv.it_factor->leaf->TensorX()), static_cast<int>(R), tnsDims[i]);
360  mv.mttkrp = mv.temp_matrix.transpose();
361  mv.cwise_factor_product = tensorToMatrix(mv.it_factor->leaf->Gramian(), static_cast<int>(R), static_cast<int>(R));
362  mv.currentFactor = tensorToMatrix(mv.it_factor->factor, tnsDims[i], static_cast<int>(R));
363 
364  update_factor(i, R, tnsDims, status, mv);
365  mv.it_factor++;
366  }
367 
368  mv.it_factor = mv.factors.end()-1;
369  mv.last_gramian = tensorToMatrix(mv.it_factor->gramian, static_cast<int>(R), static_cast<int>(R));
370  cost_function(mv, status);
371  status.rel_costFunction = status.f_value / sqrt(status.frob_tns);
372  if(status.options.normalization && !mv.all_orthogonal)
373  Normalize(mv.weight_factor, static_cast<int>(R), tnsDims, mv.factors);
374 
375  // ---- Terminating condition ----
376  if (status.ao_iter >= status.options.max_iter || status.rel_costFunction < status.options.threshold_error)
377  {
378  for(std::size_t i=0; i<TnsSize; ++i)
379  status.factors[i] = tensorToMatrix(mv.factors[i].factor, tnsDims[i], static_cast<int>(R));
380 
381  if(status.options.writeToFile)
382  writeFactorsToFile(status);
383  break;
384  }
385 
386  if (status.options.acceleration)
387  {
388  mv.norm_factors = mv.factors;
389  if (status.ao_iter > 1)
390  line_search_accel(tnsDims, R, mv, status);
391 
392  mv.old_factors = mv.norm_factors;
393  }
394  } // end of while
395  }
