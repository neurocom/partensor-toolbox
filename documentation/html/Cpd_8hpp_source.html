145  /*
146  * Compute the cost function value at the end of each outer iteration
147  * based on the last accelerated factor.
148  *

150  * @param st [in] Struct where the returned values of @c Cpd are stored.

165  void cost_function2(Member_Variables const &mv,
166  Status &st)

168  st.f_value = sqrt( st.frob_tns -2 * (mv.mttkrp.cwiseProduct(st.factors[lastFactor])).sum() +

172  /*
173  * Based on each factor's constraint, a different
174  * update function is used at every outer iteration.
175  *
176  * Computes also factor^T * factor at the end.
177  *
178  * @param idx [in] Factor to be updated.
179  * @param mv [in] Struct where ALS variables are stored.
180  * @param st [in,out] Struct where the returned values of @c Cpd are stored.
181  * Updates the @c stl array with the factors.
182  */
183  void update_factor(int const idx,
184  Member_Variables &mv,
185  Status &st )
186  {
187  // Update factor
188  switch ( st.options.constraints[idx] )
189  {
190  case Constraint::unconstrained:
191  {
192  st.factors[idx].noalias() = mv.mttkrp[idx] * mv.cwise_factor_product.inverse();
193  break;
194  }
195  case Constraint::nonnegativity:
196  {
197  mv.temp_matrix = st.factors[idx];
198  NesterovMNLS(mv.cwise_factor_product, mv.mttkrp[idx], st.options.nesterov_delta_1,
199  st.options.nesterov_delta_2, st.factors[idx]);
200  if(st.factors[idx].cwiseAbs().colwise().sum().minCoeff() == 0)
201  st.factors[idx] = 0.9 * st.factors[idx] + 0.1 * mv.temp_matrix;
202  break;
203  }
204  case Constraint::orthogonality:
205  {
206  mv.temp_matrix = mv.mttkrp[idx].transpose() * mv.mttkrp[idx];
207  Eigen::SelfAdjointEigenSolver<Matrix> eigensolver(mv.temp_matrix);
208  mv.temp_matrix.noalias() = (eigensolver.eigenvectors())
209  * (eigensolver.eigenvalues().cwiseInverse().cwiseSqrt().asDiagonal())
210  * (eigensolver.eigenvectors().transpose());
211  st.factors[idx].noalias() = mv.mttkrp[idx] * mv.temp_matrix;
212  break;
213  }
214  case Constraint::sparsity:
215  break;
216  default: // in case of Constraint::constant
217  break;
218  }
219 
220  // Compute A^T * A + B^T * B + ...
221  mv.factor_T_factor[idx].noalias() = st.factors[idx].transpose() * st.factors[idx];
222  }

224  /*
225  * @brief Line Search Acceleration
226  *
227  * Performs an acceleration step on the updated factors, and keeps the accelerated factors
228  * when the step succeeds. Otherwise, the acceleration step is ignored.
229  * Line Search Acceleration reduces the number of outer iterations in the ALS algorithm.
230  *
231  * @note This implementation ONLY, if factors are of @c Matrix type.
232  *
233  * @param mv [in,out] Struct where ALS variables are stored.
234  * In case the acceleration step is successful the Gramian
235  * matrices of factors are updated.
236  * @param st [in,out] Struct where the returned values of @c Cpd are stored.
237  * If the acceleration succeeds updates @c factors
238  * and cost function value.
239  *
240  */
241  void line_search_accel(Member_Variables &mv,
242  Status &st)
243  {
244  double f_accel = 0.0; // Objective Value after the acceleration step
245  double const accel_step = pow(st.ao_iter+1,(1.0/(st.options.accel_coeff)));
246 
247  MatrixArray accel_factors;
248  MatrixArray accel_gramians;
249 
250  for(std::size_t i=0; i<TnsSize; ++i)
251  {
252  accel_factors[i] = mv.old_factors[i] + accel_step * (st.factors[i] - mv.old_factors[i]);
253  accel_gramians[i] = accel_factors[i].transpose() * accel_factors[i];
254  }
255 
256  f_accel = accel_cost_function(mv, st, accel_factors, accel_gramians);
257  if (st.f_value > f_accel)
258  {
259  st.factors = accel_factors;
260  mv.factor_T_factor = accel_gramians;
261  st.f_value = f_accel;
262  Partensor()->Logger()->info("Acceleration Step SUCCEEDED! at iter: {}", st.ao_iter);
263  }
264  else
265  st.options.accel_fail++;
266 
267  if (st.options.accel_fail==5)
268  {
269  st.options.accel_fail=0;
270  st.options.accel_coeff++;
271  }
272  }

274  /*
275  * Sequential implementation of Alternating Least Squares (ALS) method.
276  *
277  * @param R [in] The rank of decomposition.
278  * @param mv [in] Struct where ALS variables are stored and being updated
279  * until a termination condition is true.
280  * @param st [in,out] Struct where the returned values of @c Cpd are stored.
281  */
282  void als(std::size_t const R,
283  Member_Variables &mv,
284  Status &status)
285  {
286  for (std::size_t i=0; i<TnsSize; i++)
287  {
288  mv.factor_T_factor[i].noalias() = status.factors[i].transpose() * status.factors[i];
289  mv.tns_mat[i] = Matricization(mv.tnsX, i);
290  }
291 
292  if(status.options.acceleration)
293  {
294  mv.tnsX_mat_lastFactor_T = mv.tns_mat[lastFactor].transpose();
295  }
296 
297  if(status.options.normalization)
298  {
299  choose_normilization_factor(status, mv.all_orthogonal, mv.weight_factor);
300  }
301 
302  // Normalize(static_cast<int>(R), mv.factor_T_factor, status.factors);
303  status.frob_tns = square_norm(mv.tnsX);
304  cost_function_init(mv, status);
305  status.rel_costFunction = status.f_value/sqrt(status.frob_tns);
306 
307  // ---- Loop until ALS converges ----
308  while(1)
309  {
310  status.ao_iter++;
311  Partensor()->Logger()->info("iter: {} -- fvalue: {} -- relative_costFunction: {}", status.ao_iter,
312  status.f_value, status.rel_costFunction);
313 
314  for (std::size_t i=0; i<TnsSize; i++)
315  {
316  mttkrp(status.factors, mv.tns_mat[i], i, mv.krao[i], mv.mttkrp[i]);
317  mv.cwise_factor_product = PartialCwiseProd(mv.factor_T_factor, i);
318 
319  // Update factor
320  update_factor(i, mv, status);
321 
322  // Cost function Computation
323  if(i == lastFactor)
324  cost_function(mv, status);
325  }
326 
327  status.rel_costFunction = status.f_value/sqrt(status.frob_tns);
328  if(status.options.normalization && !mv.all_orthogonal)
329  Normalize(mv.weight_factor, static_cast<int>(R), mv.factor_T_factor, status.factors);
330 
331 
332  // ---- Terminating condition ----
333  if (status.rel_costFunction < status.options.threshold_error || status.ao_iter >= status.options.max_iter)
334  {
335  if(status.options.writeToFile)
336  writeFactorsToFile(status);
337  break;
338  }
339 
340  if (status.options.acceleration)
341  {
342  mv.norm_factors = status.factors;
343  // ---- Acceleration Step ----
344  if (status.ao_iter > 1)
345  line_search_accel(mv, status);
346 
347  mv.old_factors = mv.norm_factors;
348  }
349  } // end of while
350  }
